HDU1018 - Big Number

2019-02-24

Big Number

  直接套用stirling数的公式即可。关于strling数，可参考下图

代码如下：

#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
#include <cmath>
#define INF 0x7fffffff
#define pi acos(-1.0)
#define mst(a,b) memset(a,b,sizeof(a))
const double e=2.718281828459;
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int maxn=1e6+5;
int main()
{
    int t;
//    cout<<log10(103)<<endl;
    cin>>t;
    while(t--)
    {
        double n;
        cin>>n;
        ll ans=0;
        ans=(int)(floor((log10((2*pi*n)))/2+n*log10(n/e))+1);        //stirling公式 
        cout<<ans<<endl;
    }
    return 0;
}

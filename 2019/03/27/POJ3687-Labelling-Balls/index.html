int gra[205][205];

int ans[40005];
int n,m;
bool topo()
{
    int tag;int cnt=0;
    for(int i=n;i>=1;i--){        //这样反向循环的原理到底是什么？
        for(int j=n;j>=1;j--){
            if(deg[j]==0){
                tag=j;
                deg[j]--;
                ans[j]=i;        //???
                cnt++;
                break;
            }
        }
        for(int j=1;j<=n;j++){
            if(gra[tag][j]>0)
                deg[j]--;
        }
    }
    if(cnt<n)
        return false;
    return true;
}
int main()
{
    int t;
